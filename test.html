Locale: locale-tr.js

File:

// Define catalog object
            // ----------------------------------------------------------------
            // Türkçe
            dji18njs.catalog.tr = {
                "key": 'anahtar',
                "keys": 'anahtarlar',
                "email": 'eposta',
                "emails": 'epostalar',
                "Hello world in Turkish": 'Türkçe Merhaba dünya',
                "Hello %s in Turkish": 'Türkçe Merhaba %s',
                "Hello %(user)s in Turkish": 'Türkçe Merhaba %(user)s',
                "There %(is)s %(amount)s new %(email)s for you.": 'Sizin için %(amount)s tane yeni eposta var.',
            };

Code:

var strFormat = "There %(is)s %(amount)s new %(email)s for you.";
            var intAmount = 2;
            var objWords = {
                amount: intAmount,
                is: dji18njs.ngettext("is", "are", intAmount),
                email: dji18njs.ngettext("email", "emails", intAmount),
            };

            document.write("<strong>", dji18njs.lang_id, "</strong>:", "<br/>");
            document.write(dji18njs.gettext("Hello world"), "<br/>");
            document.write(dji18njs.ngettext("key", "keys", 5), "<br/>");
            document.write(dji18njs.interpolate("Hello %s", ["vigo"]), "<br/>");
            document.write(dji18njs.interpolate("Hello %(user)s", { user: "lego" }, true), "<br/>");
            document.write(dji18njs.interpolate(strFormat, objWords, true), "<br/>");
            document.write("<hr/>");

            dji18njs.lang_id = "tr";
